import type { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  const base = "https://mementomori-psi.vercel.app"; // domeniul tÄƒu
  return {
    rules: {
      userAgent: "*",
      allow: "/",
    },
    sitemap: `${base}/sitemap.xml`,
  };
}
